Function prisma(largura, comprimento, altura)
prisma = largura * comprimento * altura
End Function

Function cilindro(raio, altura)
cilindro = (WorksheetFunction.Pi) * ((raio) ^ 2) * altura
End Function

Function esfera(raio)
esfera = (4 / 3) * (WorksheetFunction.Pi) * ((raio) ^ 3)
End Function

Function cone(raio, altura)
cone = (1 / 3) * (WorksheetFunction.Pi) * ((raio) ^ 2) * altura
End Function


Function piramide(comprimento, altura)
piramide = ((comprimento) ^ 2) * (altura) / 3
End Function
