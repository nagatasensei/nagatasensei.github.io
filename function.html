Function vprisma(largura, comprimento, altura)
vprisma = largura * comprimento * altura
End Function

Function vcubo(lado)
vcubo = (lado) ^ 3
End Function

Function vcil(raio, altura)
vcil = (WorksheetFunction.Pi) * ((raio) ^ 2) * altura
End Function

Function vesfera(raio)
vesfera = (4 / 3) * (WorksheetFunction.Pi) * ((raio) ^ 3)
End Function

Function vcone(raio, altura)
vcone = (1 / 3) * (WorksheetFunction.Pi) * ((raio) ^ 2) * altura
End Function
